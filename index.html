Easy configuration


import { Storage, Configuration } from 'typefs';

Storage.config: Configuration = {
    default: 'tmp',
    disks: {
        tmp: {
            driver: 'file',
            root: '/tmp/',
            jail: true,
        }
        app: {
            driver: 'file',
            root: '/app/',
            jail: true,
        }
        config: {
            driver: 'file',
            root: '/app/config',
            jail: true,
        }
    }
}
